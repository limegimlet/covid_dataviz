pop_url = 'https://opendata.arcgis.com/datasets/b2e6d8854d9744ca88144d30bef06a76_1.csv'

# clean kreis names
clean_kreis = [k.split(" ")[1] for k in kreis]
q = "GEN in @clean_kreis"
sh_pop = pop.query(q)[['GEN', 'EWZ']].rename(columns={'GEN':'Landkreis'}).set_index('Landkreis')
sh_df['Landkreis'] = sh_df['Landkreis'].replace(kreis, clean_kreis)

incid_kreis = sh_df.groupby(['Landkreis', 'Meldedatum'])['AnzahlFall_7T'].sum()
incid_kreis = incid_kreis.reset_index().merge(sh_pop.reset_index()).set_index(['Landkreis', 'Meldedatum'])
incid_kreis['EWZ'] = incid_kreis['EWZ'].astype('float')

#incid_kreis.loc['Steinburg']
#incid_kreis['Inzid_7T'].unstack(0).rolling(7).sum().tail(50)

incid_kreis['Inzid_7T'] = incid_kreis['AnzahlFall_7T'].multiply(100000).divide(incid_kreis['EWZ']).round(2)


# what is the 7-day rate of change?
wave2 = incid_kreis['Inzid_7T'].unstack(0).loc['2020-09-01':]

pal=cf.colors.get_scales('dflt')

for n in ints:
